commands:

- run flutter on web with custom port and browser:
flutter run -d web-server --web-hostname localhost --web-port 3000

- run on chrome
flutter run (if no emulator is opened)
flutter run -d chrome
flutter run -d chrome --web-port 3000

- build appbundle (.aab)
flutter build appbundle

- build apk (.apk)
flutter build apk

- production level apk
flutter build --release apk
